import streamlit as st

def konversi_celsius(celsius):
    """Fungsi untuk mengkonversi suhu dari Celsius ke skala lain."""
    
    # Konversi ke Fahrenheit: F = (C * 9/5) + 32
    fahrenheit = (celsius * 9/5) + 32
    
    # Konversi ke Reamur: R = C * 4/5
    reamur = celsius * 4/5
    
    # Konversi ke Kelvin: K = C + 273.15
    kelvin = celsius + 273.15
    
    # Mengembalikan hasil konversi
    return fahrenheit, reamur, kelvin

# --- Antarmuka Streamlit ---
st.set_page_config(layout="wide")
st.title("ğŸŒ¡ï¸ Kalkulator Konversi Suhu (Celsius)")
st.markdown("Masukkan nilai suhu dalam Celsius untuk mendapatkan konversi ke Fahrenheit, Reamur, dan Kelvin.")

# Input Nilai Celsius (mengganti input())
# st.number_input lebih baik daripada st.text_input untuk nilai numerik.
nilai_celsius = st.number_input(
    "Masukkan nilai suhu dalam Celsius (Â°C):",
    min_value=-273.15, # Batas bawah adalah nol mutlak
    value=25.0, # Nilai default
    step=0.1,
    format="%.2f"
)

# Panggil fungsi konversi
F, R, K = konversi_celsius(nilai_celsius)

st.subheader("Hasil Konversi")

# Menampilkan hasil (mengganti print())
# Menggunakan st.metric dalam kolom untuk tampilan yang rapi dan informatif
col1, col2, col3 = st.columns(3)

    st.metric(label="Fahrenheit (Â°F)", value=f"{F:.2f}", delta="F = (C * 9/5) + 32")

with col2:
    st.metric(label="Reamur (Â°R)", value=f"{R:.2f}", delta="R = C * 4/5")

with col3:
    st.metric(label="Kelvin (K)", value=f"{K:.2f}", delta="K = C + 273.15")
